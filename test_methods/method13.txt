        case STRING:
            return decodeSTRING(inserter, meta);

            return decodeDATA(inserter, meta);
        case ARRAY:
            return decodeARRAY(inserter, meta);
        case OBJECT:
            return decodeOBJECT(inserter, meta);

}]]],
